HJ1 字符串最后一个单词的长度

var readline = require("readline"); const rl = readline.createInterface({ input: process.stdin, output: process.stdout, }); rl.on("line", function (line) { let array = line.split(" "); console.log(array[array.length - 1].length); });

var readline = require("readline"); const rl = readline.createInterface({ input: process.stdin, output: process.stdout, }); const lines = []; rl.on("line", function (line) { lines.push(line); if (lines.length === 2) { const input = lines[0]; const target = lines[1]; var res = input.match(new RegExp(target, "gim")); if (res === null) console.log(0); else console.log(res.length); } });
